\begin{figure}[htbp]
  \centering
  \begin{tikzpicture}[
    node distance=1.5cm,
    box/.style={draw, fill=blue!10, text width=2.5cm, align=center, minimum height=1cm, rounded corners},
    worker/.style={draw, fill=green!10, text width=2cm, align=center, minimum height=0.8cm, rounded corners},
    database/.style={draw, cylinder, cylinder uses custom fill, cylinder body fill=gray!20, cylinder end fill=gray!30, shape border rotate=90, minimum width=1.5cm, minimum height=1.5cm, aspect=0.25, align=center},
    arrow/.style={-stealth, thick}
  ]

    % Components
    \node (orch) [box, fill=blue!30] {\textbf{Master Orchestrator} \\ (LangGraph)};
    \node (reasoning) [box, above=of orch] {Reasoning Suite \\ (ToT, ReWOO, Refl)};
    \node (registry) [box, below=of orch] {Worker Agent \\ Registry};
    
    % Persistence (Fixed Issue 08)
    \node (persistence) [database, right=2.5cm of orch] {State Store \\ (Redis/Pg)};

    % Specialized Workers
    \node (web) [worker, below left=1.2cm and 0.2cm of registry] {Web Agent};
    \node (auth) [worker, below=1.2cm of registry] {Auth Agent};
    \node (net) [worker, below right=1.2cm and 0.2cm of registry] {Network Agent};

    % Connections
    \draw [arrow] (orch) -- (reasoning) node[midway, right] {\small Prompt};
    \draw [arrow] (reasoning) -- (orch) node[midway, left] {\small Blueprint};
    \draw [arrow] (orch) -- (registry) node[midway, right] {\small Task};
    \draw [arrow] (orch) -- (persistence) node[midway, above] {\small Checkpoint};
    
    \draw [arrow] (registry.south) -- (web.north);
    \draw [arrow] (registry.south) -- (auth.north);
    \draw [arrow] (registry.south) -- (net.north);

    % Background Box for Trusted Zone
    \begin{scope}[on background layer]
      \node [draw, dashed, gray, fill=blue!5, fit=(orch) (reasoning) (persistence), inner sep=0.5cm] (trust) {};
      \node [below right] at (trust.north west) {\small \textit{Trusted Reasoning Zone}};
    \end{scope}

  \end{tikzpicture}
  \caption{The CMatrix system architecture. The Master Orchestrator manages the stateful DAG, persisting checkpoints to a Redis/PostgreSQL store while delegating operational tasks to a registry of specialized worker agents.}
  \label{fig:system-architecture}
\end{figure}
